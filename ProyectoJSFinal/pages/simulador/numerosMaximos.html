function encontrarMaximo(arr) {
    let max = arr[0]; // Línea 2
    for (let i = 1; i < arr.length; i++) { // Línea 3
        if (arr[i] > max) { // Línea 4
            max = arr[i];
        }
    }
    return max;
}
